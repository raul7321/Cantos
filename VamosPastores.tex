\subsection{¡Vamos pastores, vamos!}

\begin{center}

{\Large
Vamos pastores vamos,\\
vamos a Belén,\\
a ver en ese niño\\
la gloria del Edén,\\
a ver en ese niño\\
la gloria del Edén.\\
\vspace{1em}
Ese precioso niño,\\
yo me muero por él.\\
Sus ojitos me encantan,\\
su boquita también.\\
El padre lo acaricia\\
la madre mira en él,\\
y los dos extasiados,\\
contemplan aquel ser,\\
contemplan aquel ser.\\
\vspace{1em}
Vamos pastores vamos,\\
vamos a Belén,\\
a ver en ese niño\\
la gloria del Edén,\\
a ver en ese niño\\
la gloria del Edén.\\
la gloria del Edén, del Edén...\\
}
\end{center}
