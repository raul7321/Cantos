\subsection{Resucitó}

\begin{center}

{\Large
Resucitó, resucitó,\\
resucitó, ¡Aleluya!\\
Aleluya, aleluya, \\
aleluya, ¡Resucitó!\\
\vspace{1em}
La muerte,\\
dónde está la muerte,\\
dónde está mi muerte,\\
dónde su victoria.\\
\vspace{1em}
Gracias,\\
sean dadas al Padre,\\
que nos dio Su Reino,\\
donde se vive el Amor.\\
\vspace{1em}
Alegría,\\
alegría, hermanos,\\
que si hoy nos queremos,\\
es que Resucitó.}
\end{center}
