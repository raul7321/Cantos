\subsection{Fuente Bautismal}
\begin{center}

{\Large
Fuente Bautismal de donde brota la fe,\\
ruta de la luz, camino de salvación.\\
\vspace{2ex}
Hijos de Dios por la gracia, \\
miembros de Cristo en su Iglesia.\\
\vspace{2ex}
Fuente Bautismal de donde brota la fe,\\
ruta de la luz, camino de salvación.\\
\vspace{2ex}
Todos unidos en Cristo,\\
todos formamos su Cuerpo.\\
\vspace{2ex}
Fuente Bautismal de donde brota la fe,\\
ruta de la luz, camino de salvación.\\
\vspace{2ex}
Hoy te recibe la Iglesia,\\
en su familia cristiana.\\
\vspace{2ex}
Fuente Bautismal de donde brota la fe,\\
ruta de la luz, camino de salvación.\\
}
\end{center}
